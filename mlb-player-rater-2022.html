V-Score Calculation:
HittingScore = (avg * atBats)
                    + (3 * runs)
                    + (12 * homeRuns)
                    + (3.5 * rbis)
                    + (6 * stolenBases)
PitchingScore = (5 * strikeOuts)
                    + (3 * inningsPitched)
                    + (10 * wins)
                    + (15 * saves)
                    - (4 * baseOnBalls)
                    - (3 * hits)
                    - (4 * earnedRuns)
RotoScore = HittingScore + PitchingScore
vScore = zScore(RotoScore, position)
